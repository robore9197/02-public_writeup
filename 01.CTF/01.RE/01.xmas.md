# 5. Viết chương trình giải mã vùng nhớ
```
#include <stdio.h>
int main(){
 long constant = 926102321;
 char target[32] = {105,126,114,100,74,68,0,104,70,90,6,95,110,70,108,90,2,65,65,78,110,80,91,69,88,6,4,90,80,6,78};
 char tmp;
 for (int i = 0; i < 32; i++){
  long v3 = constant >>  (8 * (i % 4));
  for (int j = 0; j< 255; j++){
    tmp = j ^v3;
    if (tmp == target[i]){
     printf("%c",j);
    }
  }
 }
 return 0;
}
```
